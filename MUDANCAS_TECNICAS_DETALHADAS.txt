emprestarLivro() {
    // Validação 1: Dados básicos não vazios
    if (biblioteca.getUsuarios().isEmpty() || biblioteca.getLivros().isEmpty())
        return;
    
    // Validação 2: Email fornecido
    String emailUsuario = scanner.nextLine().trim();
    
    // Validação 3: Usuário existe
    Usuario usuario = biblioteca.buscarUsuarioPorEmail(emailUsuario);
    if (usuario == null)
        return;
    
    // Validação 4: ISBN fornecido
    String isbn = scanner.nextLine().trim();
    
    // Validação 5: Livro existe
    Livro livro = biblioteca.buscarPorIsbn(isbn);
    if (livro == null)
        return;
    
    // Validação 6: Livro está disponível
    if (!livro.isDisponibilidade())
        return;
    
    // Tudo OK: Emprestar
    usuario.emprestarLivro(livro);
}

⚠️ TOTAL: 6 validações antes de executar operação crítica
