private static void registrarLivroPorIsbn() {
    System.out.println("\n--- REGISTRAR LIVRO POR ISBN ---");
    System.out.print("Digite o ISBN do livro (ex: 978-8535929935): ");
    String isbn = scanner.nextLine().trim();
    
    // ✅ Validação 1: ISBN não vazio
    if (isbn.isEmpty()) {
        System.out.println("❌ ISBN não pode ser vazio!");
        return;
    }
    
    // ✅ Validação 2: ISBN não duplicado
    if (biblioteca.buscarPorIsbn(isbn) != null) {
        System.out.println("❌ Livro com este ISBN já está cadastrado!");
        return;
    }
    
    try {
        // ✅ Try-catch para erros de rede/API
        APIBuscadorDelivro.LivroTemp livroTemp = 
            APIBuscadorDelivro.buscarEConfirmarLivro(isbn);
        
        if (livroTemp != null) {
            Livro livro = new Livro(
                livroTemp.titulo,
                livroTemp.autor,
                livroTemp.isbn,
                livroTemp.ano,
                livroTemp.editora
            );
            biblioteca.adicionarLivro(livro);
        }
    } catch (Exception e) {
        // ✅ Tratamento de erro com mensagem clara
        System.out.println("❌ Erro ao processar ISBN: " + e.getMessage());
    }
}
